/// Image, pixel data stored in CPU memory (RAM)
class Image extends NativeWrapper<_Image> {
	_Image get ref => pointer.ref;
	set ref(_Image value) => pointer.ref = value;

	/// Image raw data
	Pointer<Void> get data => ref.data;
	Pointer<_Image> get test => ref.test;
	_Image get what => ref.what;
	/// Image base width
	int get width => ref.width;
	/// Image base height
	int get height => ref.height;
	/// Mipmap levels, 1 by default
	int get mipmaps => ref.mipmaps;
	/// Data format (PixelFormat type)
	int get format => ref.format;

	set data(Pointer<Void> value) => ref.data = value;
	set test(Image value) => ref.test = value.pointer;
	set what(Image value) => ref.what = value.ref;
	set width(int value) => ref.width = value;
	set height(int value) => ref.height = value;
	set mipmaps(int value) => ref.mipmaps = value;
	set format(int value) => ref.format = value;

	// ignore: unused_element_parameter, unused_element
	Image._Encapsulate(super.pointer,{ super.IsOwner, super.length }): super.fromAddress() {
		if (IsOwner)
			_nativeFinalizer.attach(this, pointer, detach: this);
	}

	// ignore: unused_element_parameter, unused_element
	Image._Recieve(_Image result): super(sizeOf<_Image>()) {
		ref = result;
		_nativeFinalizer.attach(this, pointer, detach: this);
	}

	// ==================================== Constructor =====================================

	factory Image(Pointer<Utf8> fileName) {
		_Image result = _exportImageAsCode(fileName);

		Image image = Image._Recieve(result);
		return image;
	}

	// ================================= Named Constructor ==================================

	/// Unload image from CPU memory (RAM)
	factory Image.LoadRaw(Pointer<Utf8> fileName, int width, int height, int format, int headerSize) {
		_Image result = _loadImageRaw(fileName, width, height, format, headerSize);
		Image image = Image._Recieve(result);
		return image;
	}

	/// Unload image from CPU memory (RAM)
	factory Image.LoadAnim(Pointer<Utf8> fileName, Pointer<Int32> frames) {
		_Image result = _loadImageAnim(fileName, frames);
		Image image = Image._Recieve(result);
		return image;
	}

	/// Unload image from CPU memory (RAM)
	factory Image.LoadAnimFromMemory(Pointer<Utf8> fileType, Pointer<Uint8> fileData, int dataSize, Pointer<Int32> frames) {
		_Image result = _loadImageAnimFromMemory(fileType, fileData, dataSize, frames);
		Image image = Image._Recieve(result);
		return image;
	}

	/// Unload image from CPU memory (RAM)
	factory Image.LoadFromMemory(Pointer<Utf8> fileType, Pointer<Uint8> fileData, int dataSize) {
		_Image result = _loadImageFromMemory(fileType, fileData, dataSize);
		Image image = Image._Recieve(result);
		return image;
	}

	/// Unload image from CPU memory (RAM)
	factory Image.LoadFromTexture(Texture2D texture) {
		_Image result = _loadImageFromTexture(texture.ref);
		Image image = Image._Recieve(result);
		return image;
	}

	/// Unload image from CPU memory (RAM)
	factory Image.LoadFromScreen() {
		_Image result = _loadImageFromScreen();
		Image image = Image._Recieve(result);
		return image;
	}

	// ====================================== Methods =======================================

	/// Export image to memory buffer
	Pointer<Uint8> ExportToMemory(Pointer<Utf8> fileType, Pointer<Int32> fileSize) => _exportImageToMemory(ref, fileType, fileSize);
	/// Clear image background with given color
	void ClearBackground(Color color) => _imageClearBackground(pointer, color.ref);
	/// Draw pixel within an image
	void DrawPixel(int posX, int posY, Color color) => _imageDrawPixel(pointer, posX, posY, color.ref);
	/// Draw pixel within an image (Vector version)
	void DrawPixelV(Vector2 position, Color color) => _imageDrawPixelV(pointer, position.ref, color.ref);
	/// Draw line within an image
	void DrawLine(int startPosX, int startPosY, int endPosX, int endPosY, Color color) => _imageDrawLine(pointer, startPosX, startPosY, endPosX, endPosY, color.ref);
	/// Draw line within an image (Vector version)
	void DrawLineV(Vector2 start, Vector2 end, Color color) => _imageDrawLineV(pointer, start.ref, end.ref, color.ref);
	/// Draw a line defining thickness within an image
	void DrawLineEx(Vector2 start, Vector2 end, int thick, Color color) => _imageDrawLineEx(pointer, start.ref, end.ref, thick, color.ref);
	/// Check if an image is valid (data and parameters)
	bool IsValid() => _isImageValid(ref);
	/// Export image data to file, returns true on success
	bool Export(Pointer<Utf8> fileName) => _exportImage(ref, fileName);
	/// Export image as code file defining an array of bytes, returns true on success
	bool ExportAsCode(Pointer<Utf8> fileName) => _exportImageAsCode(ref, fileName);

	// =================================== Deconstructor ====================================

	static final _nativeFinalizer = NativeFinalizer(_freeImage);

	// Unload image from CPU memory (RAM)
	@override
	Free() {
		_unloadImage(_Image);
		_nativeFinalizer.detach(this);
		super.Free();
	}
